<script>
// 生徒リスト
var STUDENTS=[[1,'芥 伊純'],[2,'石川 瑠唯'],[3,'伊藤 正義'],[4,'植松 莉菜'],[5,'牛窪 杏梨'],[6,'宇都 彩葉'],[7,'遠藤 央祐'],[8,'太田 栞'],[9,'岡本 創'],[10,'加藤 すず'],[11,'金子 恵琳'],[12,'樺山 玲美'],[13,'苅部 良'],[14,'工藤 幸希'],[15,'倉田 美桜'],[16,'坂本 陽菜'],[17,'佐藤 聖純'],[18,'島津 立'],[19,'新宮 悠空'],[20,'鈴木 惇也'],[21,'銭 美希'],[22,'田山 聖来'],[23,'槻舘 可奈'],[24,'中澤 勇明'],[25,'長橋 瀬奈'],[26,'中村 知世'],[27,'中山 歩翔'],[28,'長谷川 樹'],[29,'福田 蒼太'],[30,'福本 秀太'],[31,'藤原 幸柚'],[32,'星野 実紗希'],[33,'真中 杏依'],[34,'宮﨑 偉'],[35,'宮村 龍'],[36,'村澤 徹哉'],[37,'吉野 みやび'],[38,'渡部 颯太'],[39,'渡部 菜緒']];
var SUBJECTS=['現代','言語','地理','歴史','数I','数A','化学','物','体育','保健','芸','コミ','論','情','探求','LHR'];
var DAYS=['日','月','火','水','木','金','土'];
var TT={月:{1:'現代',2:'化学',3:'コミ',4:'歴史',5:'体育',6:'情'},火:{1:'芸',2:'芸',3:'数A',4:'論',5:'現代',6:'数I',7:'LHR'},水:{1:'地理',2:'言語',3:'歴史',4:'数I',5:'数A',6:'探求',7:'探求'},木:{1:'言語',2:'化学',3:'情',4:'地理',5:'体育',6:'物'},金:{1:'コミ',2:'物',3:'論',4:'保健',5:'数I',6:'コミ'}};
var REASONS=['出席','欠席','出席停止','遅刻','早退','保健室','その他'];

var periodCount=6, periodData={}, reason='出席', selectedDow='';

window.onload = function(){
  var today=new Date();
  var yyyy=today.getFullYear(), mm=String(today.getMonth()+1).padStart(2,'0'), dd=String(today.getDate()).padStart(2,'0');
  document.getElementById('dateInput').value=yyyy+'-'+mm+'-'+dd;
  onDateChange();
};

function onDateChange(){
  var val=document.getElementById('dateInput').value;
  if(!val)return;
  var d=new Date(val.replace(/-/g,'/')+' 00:00:00');
  selectedDow=DAYS[d.getDay()];
  document.getElementById('dateDisplay').textContent=d.getFullYear()+'年'+(d.getMonth()+1)+'月'+d.getDate()+'日（'+selectedDow+'）';
  document.getElementById('dateDisplay').style.display='block';
  periodData={};
  var tt=TT[selectedDow]||{};
  periodCount=Object.keys(tt).length||6;
  document.getElementById('periodVal').textContent=periodCount;
  for(var p=1;p<=8;p++) periodData[p]={subj:tt[p]||'',state:tt[p]?0:2};
  renderTimetable(); updateReasonSection(); updateOutput();
}

function changeNo(d){
  var inp=document.getElementById('noInput');
  var next=Math.min(39,Math.max(1,(parseInt(inp.value)||0)+d));
  inp.value=next; onNoChange();
}

function onNoChange(){
  var no=parseInt(document.getElementById('noInput').value), s=null;
  for(var i=0;i<STUDENTS.length;i++) if(STUDENTS[i][0]===no){ s=STUDENTS[i]; break; }
  var el=document.getElementById('nameDisplay');
  if(s){ el.textContent=no+'番：'+s[1]; el.style.display='block'; } else el.style.display='none';
  updateOutput();
}

function changePeriod(d){
  var n=periodCount+d; if(n<1||n>8)return;
  periodCount=n; document.getElementById('periodVal').textContent=n;
  for(var p=1;p<=periodCount;p++) if(!periodData[p]) periodData[p]={subj:'',state:0};
  renderTimetable(); updateOutput();
}

function renderTimetable(){
  var html='';
  for(var p=1;p<=periodCount;p++){
    var d=periodData[p]||(periodData[p]={subj:'',state:0});
    var btnCls='attend-btn'+(d.state===1?' absent':d.state===2?' none':'');
    var btnLabel=d.state===0?'○参加':d.state===1?'×不参加':'―なし';
    var opts='<option value="">科目</option>'+SUBJECTS.map(function(s){return '<option value="'+s+'"'+(d.subj===s?' selected':'')+'>'+s+'</option>';}).join('');
    html+='<div class="period-row"><div class="period-num">'+p+'</div><select class="subj-sel" onchange="setSubj('+p+',this.value)">'+opts+'</select><button class="'+btnCls+'" onclick="cycleState('+p+')">'+btnLabel+'</button></div>';
  }
  document.getElementById('periodList').innerHTML=html;
  document.getElementById('timetableCard').style.display='block';
  document.getElementById('reasonCard').style.display='block';
}

function setSubj(p,v){ periodData[p].subj=v; updateOutput(); }

function cycleState(p){
  periodData[p].state=(periodData[p].state+1)%3;
  var hasAbs=false; for(var k in periodData) if(periodData[k].state===1) hasAbs=true;
  if(hasAbs && reason==='出席') reason='欠席'; // 不参加があれば自動で欠席に
  renderTimetable(); updateReasonSection(); updateOutput();
}

function updateReasonSection(){
  var html=REASONS.map(function(r){return '<button class="reason-btn'+(r===reason?' active':'')+'" onclick="setReason(\''+r+'\')">'+r+'</button>';}).join('');
  document.getElementById('reasonBtns').innerHTML=html;
}

function setReason(r){ reason=r; updateReasonSection(); updateOutput(); }

function updateOutput(){
  var no=document.getElementById('noInput').value, dateVal=document.getElementById('dateInput').value;
  if(!dateVal||!no){ document.getElementById('outputCard').style.display='none'; return; }
  var d=new Date(dateVal.replace(/-/g,'/')+' 00:00:00'), dateStr=d.getFullYear()+'/'+(d.getMonth()+1)+'/'+d.getDate();
  var note=document.getElementById('noteInput').value, counts={};
  SUBJECTS.forEach(function(s){counts[s]=0;});
  for(var p=1;p<=periodCount;p++){ var pd=periodData[p]; if(pd&&pd.state===1&&pd.subj) counts[pd.subj]++; }
  var row=[reason,dateStr,selectedDow].concat(SUBJECTS.map(function(s){return counts[s]||0;})).concat([note]);
  document.getElementById('outputBox').textContent=row.join(' | ');
  document.getElementById('outputCard').dataset.tsv=row.join('\t');
  document.getElementById('outputCard').style.display='block';
}

function doCopy(){
  var text=document.getElementById('outputCard').dataset.tsv||'', btn=document.getElementById('copyBtn');
  var ta=document.createElement('textarea'); ta.value=text; document.body.appendChild(ta); ta.select();
  try{ document.execCommand('copy'); btn.textContent='✓ コピーしました！'; document.getElementById('copyHint').style.display='block'; }catch(e){}
  document.body.removeChild(ta);
}
